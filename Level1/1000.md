```jsx
const readline = require("readline");
const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout
});

rl.on("line", (input) => {
  const inputValue = input.split(" ");
  const result = Number(inputValue[0]) + Number(inputValue[1]);
  console.log(result);
  rl.close();
});
```
