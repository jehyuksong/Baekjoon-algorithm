```jsx
const readline = require("readline");
const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout
});

rl.on("line", (line) => {
  const input = line.split(" ");
  let [a, b, c] = input;
  a = Number(a);
  b = Number(b);
  c = Number(c);
  console.log(a, b, c);
  console.log((a + b) % c);
  console.log(((a % c) + (b % c)) % c);
  console.log((a * b) % c);
  console.log(((a % c) * (b % c)) % c);
});
```
