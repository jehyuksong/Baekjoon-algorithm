const readline = require("readline");
const rl = readline.createInterface({
  input: process.stdin,
  outpt: process.stdout
});

rl.on("line", input => {
  const inputs = input.split(" ");
  let hour = Number(inputs[0]);
  let minute = Number(inputs[1]);

  if (minute < 45) {
    minute += 15;
    hour -= 1;
    if (hour === -1) {
      hour += 24;
    }
  } else {
    minute -= 45;
  }

  console.log(hour, minute);

  rl.close();
});
