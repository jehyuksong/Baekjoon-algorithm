// 2-3 (2753)
const readline = require("readline");
const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout
});

rl.on("line", input => {
  const year = Number(input);

  if (year % 4 === 0) {
    if (year % 100 === 0) {
      if (year % 400 === 0) {
        console.log(1);
      } else {
        console.log(0);
      }
    } else {
      console.log(1);
    }
  } else {
    console.log(0);
  }
  rl.close();
});
