```jsx
// 3-9 (2438)
const readline = require("readline");
const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout
});

rl.on("line", input => {
  const n = parseInt(input);
  let result = "";

  for (i = 1; i <= n; i++) {
    result += "*";
    console.log(result);
  }
  rl.close();
});
```
