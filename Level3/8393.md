```jsx
// 3-3 (8393)
const readline = require("readline");
const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout
});

rl.on("line", input => {
  let result = 0;
  const n = Number(input);

  for (i = 1; i <= n; i++) {
    result = result + i;
  }

  console.log(result);

  rl.close();
});
```
