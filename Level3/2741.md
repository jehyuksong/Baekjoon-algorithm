```jsx
// 3-5 (2741)
const readline = require("readline");
const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout
});

let result = "";

rl.on("line", input => {
  const n = parseInt(input);
  for (i = 1; i <= n; i++) {
    result += i + "\n";
  }
  console.log(result);
  rl.close();
}).on("close", () => {
  process.exit();
});
```
