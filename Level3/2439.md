// 3-10 (2439)
const readline = require("readline");
const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout
});

rl.on("line", input => {
  const n = parseInt(input);
  let result = "";

  for (i = 0; i < n; i++) {
    for (j = n - 1; j > i; j--) {
      result += " ";
    }
    for (k = 0; k <= i; k++) {
      result += "*";
    }
    result += "\n";
  }
  console.log(result);
});
