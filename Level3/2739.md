```jsx
// 3-1 (2739)
const readline = require("readline");
const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout
});

rl.on("line", input => {
  const n = Number(input);
  for (i = 1; i < 10; i++) {
    console.log(`${n} * ${i} = ${n * i}`);
  }

  rl.close();
});
```
