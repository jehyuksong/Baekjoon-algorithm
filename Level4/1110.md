// 4-3 (1110)

const readline = require("readline");
const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout
});

let num = "";
let change = "";
let notSame = true;

rl.on("line", input => {
  if (input.length === 1) {
    num = `0${input}`;
  } else {
    num = input;
  }

  let front = num[1];
  let back = String(parseInt(num[0]) + parseInt(num[1]));
  if (back.length === 2) {
    change = front + back[1];
  } else {
    change = front + back;
  }

  let count = 1;

  while (notSame) {
    if (num === change) {
      notSame = false;
    } else {
      front = change[1];
      back = String(parseInt(change[0]) + parseInt(change[1]));
      if (back.length === 2) {
        change = front + back[1];
      } else {
        change = front + back;
      }
      count++;
    }
  }
  console.log(count);
});
